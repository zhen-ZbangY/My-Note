1.创建核心shell脚本--》核心代码
[root@docker busybox]# vim while.sh
#!/bin/sh

i=1
while true
do
        echo  "hello,world,welcome to sanchuang $i"
        sleep  1
        let i++
done  
2.创建Dockerfile文件
[root@docker busybox]# vim Dockerfile

FROM  busybox:latest
WORKDIR  /
COPY  .  /
RUN  touch sc.txt && mkdir sanchuang && sleep 5
ENTRYPOINT ["/bin/sh","/while.sh"]

=========

FROM  busybox:latest  #指定基础镜像
WORKDIR  /  #指定进入容器内部会进入的目录---》家目录
COPY  .  /  #复制宿主机当前目录下的所有的文件和文件夹到容器里的系统的根目录下
RUN  touch sc.txt && mkdir sanchuang && sleep 5  #当我们制作镜像的时候，会启动一个临时的容器，在这个临时的容器里进行操作
                                                  #只是在制作镜像的时候，基于基础镜像安装或者新建某些文件和文件夹等
ENTRYPOINT ["/bin/sh","/while.sh"]  #指定镜像制作完成后，启动容器的时候运行的第1个命令  /bin/sh  /while.sh

==========
